\subsection{Weight functional and positivity}
  \label{subsec:weight-functional-and-positivity}

  Edge weights are parameterized by a positive connectivity (stiffness)
  matrix $K_{uv}>0$:
  \[
    w_{uv} = \frac{1}{K_{uv}}.
  \]
  To avoid circularity, $K_{uv}$ is taken as a functional of the
  coarse-grained background $\bar{\chi}$, not of $\chi$ itself. We use
  \begin{equation}
    w_{uv}(\bar{\chi})
    =
    \frac{1}{K_0}
    \left[
      1 + \left(\frac{\bar{\chi}_u-\bar{\chi}_v}{\chi_c}\right)^2
    \right],
    \qquad
    K_{uv} = \frac{1}{w_{uv}}.
    \label{eq:weights}
  \end{equation}
  Positivity is manifest, hence $d_{ij}^{W}$ defines a weighted
  path metric on any connected graph.

  \paragraph{Intrinsic saturation scale.}
    The scale $\chi_c$ is defined intrinsically by the edge-variance
    \begin{equation}
      \chi_c^2
      =
      \frac{1}{|E|}\sum_{(u,v)\in E}(\bar{\chi}_u - \bar{\chi}_v)^2
      -
      \left(\frac{1}{|E|}\sum_{(u,v)\in E}(\bar{\chi}_u - \bar{\chi}_v)\right)^2,
      \label{eq:chi_c_intrinsic}
    \end{equation}
    which depends only on the combinatorial structure and $\bar{\chi}$.
    It is translation-invariant, dimensionally consistent, and strictly
    positive whenever non-trivial contrasts exist.

  \paragraph{Rescaling invariance.}
    \begin{lemma}
      In the admissible spectral regime selected by $F_{\lambda^*}$,
      the normalized spectral distance $d_S(i,j)$ is invariant to leading
      order under $\chi_c \mapsto \alpha\,\chi_c$.
    \end{lemma}

    \begin{proof}[Sketch]
For $|\bar{\chi}_u-\bar{\chi}_v|/\chi_c \ll 1$,
\[
  w_{uv}
  =
  K_0^{-1}
  \Big[1 + \mathcal{O}((\bar{\chi}_u-\bar{\chi}_v)^2/\chi_c^2)\Big].
\]
The leading Laplacian is therefore $\chi_c$-independent.
Rescaling affects only higher-order terms suppressed by
$F_{\lambda^*}$, leaving $d_S$ unchanged at leading order.
    \end{proof}
